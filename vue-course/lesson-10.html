Итоговый проект — Todo App

Цель проекта

Создать полноценное приложение "Список задач" (Todo App), используя все изученные технологии Vue.js.

Функционал

  • ✅ Добавление задачи
  • ✅ Удаление задачи
  • ✅ Отметка выполнено/не выполнено
  • ✅ Фильтрация (все/активные/выполненные)
  • ✅ Счётчик активных задач
  • ✅ Сохранение в localStorage

  • Vue Router (страницы: список, архив)
  • Pinia (глобальное состояние)

Пример Store (Pinia)

import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export const useTodoStore = defineStore('todo', () => {
  const todos = ref(JSON.parse(localStorage.getItem('todos')) || [])
  
  const activeCount = computed(() => 
    todos.value.filter(t => !t.completed).length
  )
  
  function addTodo(text) {
    todos.value.push({
      id: Date.now(),
      text,
      completed: false
    })
    save()
  }
  
  function toggleTodo(id) {
    const todo = todos.value.find(t => t.id === id)
    if (todo) todo.completed = !todo.completed
    save()
  }
  
  function deleteTodo(id) {
    todos.value = todos.value.filter(t => t.id !== id)
    save()
  }
  
  function save() {
    localStorage.setItem('todos', JSON.stringify(todos.value))
  }
  
  return { todos, activeCount, addTodo, toggleTodo, deleteTodo }
})
